export const cleanCache = (opt: { all?: boolean }) => {
    const { all = true } = opt;
    if (all) {
        // remove all caches
        if (window.caches) {
            caches.keys().then((keys) => {
                keys.forEach((key) => {
                    caches.delete(key).then(() => {
                        console.log('all caches cleaned');
                    });
                });
            });
        }
    } else {
        // remove key caches
    }
};
